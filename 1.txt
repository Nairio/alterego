								stepIndex=prevs.pop();
								showStep()
							});
							nextButton=$("<button class='btn btn-danger'/>").html("Далее <span class='material-icons'>arrow_forward</span>").appendTo($("<div/>").appendTo(footer)).on("click", function(e){
								e.stopPropagation();
								var step=steps[stepIndex];
								for(var i in step.answers){
									if(step.answers[i].value){
										prevs.push(stepIndex);
										stepIndex=step.answers[i].nextStep();
										showStep();
										return;
									}
								}
								prevs.push(stepIndex);
								stepIndex=step.nextStep();
								showStep();
							});
						});
						finishData=data;
						return steps.length;
					},

				function showStep(){
					var progress=Math.round(stepIndex/steps.length*100);
					$(target).html("");

					$(nextButton).toggle(stepIndex<steps.length);
					$(prevButton).toggle(stepIndex>0);

					if(stepIndex<steps.length){
						var step=steps[stepIndex];
						$("<h2/>").html(step.question).appendTo(target);
						$("<small/>").html("Готово: "+progress+"%").appendTo(target);
						$("<div class='progress'><div class='progress-bar progress-bar-striped progress-bar-animated bg-success' role='progressbar' aria-valuenow='"+progress+"' aria-valuemin='0' aria-valuemax='100' style='width: "+progress+"%'></div></div>").appendTo(target);
						var row=$("<div class='row mt-4'/>").appendTo(target);
						step.answers.forEach(function(answer, i){
							if(answer.type=="radio"){
								$("<div class='col-sm-6 p-3'><label for='step-"+stepIndex+"-"+i+"' class='form-control ' style='cursor:pointer'><div class='custom-control custom-radio custom-control-inline'><input "+(answer.value?"checked":"")+" type='radio' id='step-"+stepIndex+"-"+i+"' name='step-"+stepIndex+"' class='custom-control-input'><div class='custom-control-label'>"+answer.title+"</div></div></label></div>").appendTo(row).on("click", "[type=radio]", function(){
									step.answers.forEach(function(a){a.value=false});
									answer.value=true;
									$(nextButton).trigger("click")
								});
							}
							if(answer.type=="radio-input"){
								$("<div class='col-sm-6 p-3'><label for='step-"+stepIndex+"-"+i+"' class='form-control pr-0' style='cursor:pointer'><div class='custom-control custom-radio custom-control-inline'><input "+(answer.value?"checked":"")+" type='radio' id='step-"+stepIndex+"-"+i+"' name='step-"+stepIndex+"' class='custom-control-input'><div class='custom-control-label'><div class='row'><div class='col text-nowrap'>"+answer.title+"</div><div class='col pr-0'><input type='text' value='"+(answer.value?answer.value:"")+"' class='form-control' style='margin-top:-0.45rem;border-right: none;border-radius: 0;'></div></div></div></div></label></div>").appendTo(row).on("change", "[type=text]", function(){
									step.answers.forEach(function(a){a.value=false});
									answer.value=$(this).val();
								}).on("click", "label", function(){$(this).find("[type='text']").focus()}).on("click", "[type=text]", function(){$(this).closest("label").trigger("click")});
							}
							if(answer.type=="checkbox-input"){
								$("<div class='col-sm-6 p-3'><label for='step-"+stepIndex+"-"+i+"' class='form-control pr-0' style='cursor:pointer'><div class='custom-control custom-checkbox custom-control-inline'><input "+(answer.value?"checked":"")+" type='checkbox' id='step-"+stepIndex+"-"+i+"' name='step-"+stepIndex+"' class='custom-control-input'><div class='custom-control-label'><div class='row'><div class='col text-nowrap'>"+answer.title+"</div><div class='col pr-0'><input type='text' value='"+(answer.value?answer.value:"")+"' class='form-control' style='margin-top:-0.45rem;border-right: none;border-radius: 0;'></div></div></div></div></label></div>").appendTo(row).on("change", "[type=text]", function(){
									answer.value=$(this).val();
								}).on("click", "label", function(){$(this).find("[type='text']").focus()}).on("click", "[type=text]", function(){$(this).closest("label").trigger("click")});

									answer.value=$(this).val();

								for(var j in steps[i].answers){
									if(steps[i].answers[j].value){
										titles.push((typeof steps[i].answers[j].value=="boolean"?steps[i].answers[j].title:steps[i].answers[j].value));
									}
								};

							}

									$(target).closest(".modal").modal("hide")
								});
							});
						});
						console.log(steps);
